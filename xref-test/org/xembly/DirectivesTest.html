55      /**
56       * Directives can make an XML document.
57       * @throws Exception If some problem inside
58       */
59      @Test
60      public void makesXmlDocument() throws Exception {
61          MatcherAssert.assertThat(
62              XhtmlMatchers.xhtml(
63                  new Xembler(
64                      new Directives()
65                          .pi("xml-stylesheet", "none")
66                          .add("page")
67                          .attr("the-name", "with \u20ac")
68                          .add("child-node").set(" the text\n").up()
69                          .add("big_text").cdata("<<hello\n\n!!!>>").up()
70                  ).xml()
71              ),
72              XhtmlMatchers.hasXPaths(
73                  "/page[@the-name]",
74                  "/page/big_text[.='<<hello\n\n!!!>>']"
75              )
76          );
77      }

124     /**
125      * Directives can add map of values.
126      * @throws Exception If some problem inside
127      * @since 0.8
128      */
129     @Test
130     public void addsMapOfValues() throws Exception {
131         final Document dom = DocumentBuilderFactory.newInstance()
132             .newDocumentBuilder().newDocument();
133         dom.appendChild(dom.createElement("root"));
134         new Xembler(
135             new Directives().xpath("/root").add(
136                 new ArrayMap<String, Object>()
137                     .with("first", 1)
138                     .with("second", "two")
139             ).add("third")
140         ).apply(dom);
141         MatcherAssert.assertThat(
142             XhtmlMatchers.xhtml(dom),
143             XhtmlMatchers.hasXPaths(
144                 "/root/first[.=1]",
145                 "/root/second[.='two']",
146                 "/root/third"
147             )
148         );
149     }

163     /**
164      * Directives can build a correct modification programme.
165      * @throws Exception If some problem inside
166      */
167     @Test
168     public void performsFullScaleModifications() throws Exception {
169         final String script = new Directives()
170             // @checkstyle MultipleStringLiteralsCheck (1 line)
171             .add("html").attr("xmlns", "http://www.w3.org/1999/xhtml")
172             .add("body")
173             .add("p")
174             .set("\u20ac \\")
175             .toString();
176         final Document dom = DocumentBuilderFactory.newInstance()
177             .newDocumentBuilder().newDocument();
178         new Xembler(new Directives(script)).apply(dom);
179         MatcherAssert.assertThat(
180             XhtmlMatchers.xhtml(dom),
181             XhtmlMatchers.hasXPaths(
182                 "/xhtml:html",
183                 "/xhtml:html/body/p[.='\u20ac \\']"
184             )
185         );
186     }

188     /**
189      * Directives can copy an existing node.
190      * @throws Exception If some problem inside
191      * @since 0.13
192      */
193     @Test
194     public void copiesExistingNode() throws Exception {
195         final Document dom = DocumentBuilderFactory.newInstance()
196             .newDocumentBuilder().newDocument();
197         final Iterable<Directive> copy = Directives.copyOf(
198             new XMLDocument(
199                 StringUtils.join(
200                     "<jeff name='Jeffrey'><first/><second/>",
201                     "<?some-pi test?>",
202                     "<file a='x'><f><name>\u20ac</name></f></file>",
203                     "<!-- some comment -->",
204                     "<x><![CDATA[hey you]]></x>  </jeff>"
205                 )
206             ).node()
207         );
208         // @checkstyle MagicNumber (1 line)
209         MatcherAssert.assertThat(copy, Matchers.iterableWithSize(19));
210         new Xembler(new Directives().add("dudes").append(copy)).apply(dom);
211         MatcherAssert.assertThat(
212             XhtmlMatchers.xhtml(dom),
213             XhtmlMatchers.hasXPaths(
214                 "/dudes/jeff[@name = 'Jeffrey']",
215                 "/dudes/jeff[first and second]",
216                 "/dudes/jeff/file[@a='x']/f[name='\u20ac']"
217             )
218         );
219     }

270      * @since 0.15.2

272     @Test
273     public void convertsToString() throws Exception {
274         final Directives dirs = new Directives();
275         for (int idx = 0; idx < Tv.TEN; ++idx) {
276             dirs.add("HELLO");
277         }
278         MatcherAssert.assertThat(
279             dirs,
280             Matchers.hasToString(Matchers.containsString("8:"))
281         );
282         MatcherAssert.assertThat(
283             new Directives(dirs.toString()),
284             Matchers.not(Matchers.emptyIterable())
285         );
286     }
