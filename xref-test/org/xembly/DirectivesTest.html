54      @Test
55      void makesXmlDocument() throws Exception {
56          MatcherAssert.assertThat(
57              XhtmlMatchers.xhtml(
58                  new Xembler(
59                      new Directives()
60                          .pi("xml-stylesheet", "none")
61                          .add("page")
62                          .attr("the-name", "with \u20ac")
63                          .add("child-node").set(" the text\n").up()
64                          .add("big_text").cdata("<<hello\n\n!!!>>").up()
65                  ).xml()
66              ),
67              XhtmlMatchers.hasXPaths(
68                  "/page[@the-name]",
69                  "/page/big_text[.='<<hello\n\n!!!>>']"
70              )
71          );
72      }

109     @Test
110     void addsMapOfValues() throws Exception {
111         final Document dom = DocumentBuilderFactory.newInstance()
112             .newDocumentBuilder().newDocument();
113         dom.appendChild(dom.createElement("root"));
114         new Xembler(
115             new Directives().xpath("/root").add(
116                 new ArrayMap<String, Object>()
117                     .with("first", 1)
118                     .with("second", "two")
119             ).add("third")
120         ).apply(dom);
121         MatcherAssert.assertThat(
122             XhtmlMatchers.xhtml(dom),
123             XhtmlMatchers.hasXPaths(
124                 "/root/first[.=1]",
125                 "/root/second[.='two']",
126                 "/root/third"
127             )
128         );
129     }

139     @Test
140     void performsFullScaleModifications() throws Exception {
141         final String script = new Directives()
142             .add("html").attr("xmlns", "http://www.w3.org/1999/xhtml")
143             .add("body")
144             .add("p")
145             .set("\u20ac \\")
146             .toString();
147         final Document dom = DocumentBuilderFactory.newInstance()
148             .newDocumentBuilder().newDocument();
149         new Xembler(new Directives(script)).apply(dom);
150         MatcherAssert.assertThat(
151             XhtmlMatchers.xhtml(dom),
152             XhtmlMatchers.hasXPaths(
153                 "/xhtml:html",
154                 "/xhtml:html/body/p[.='\u20ac \\']"
155             )
156         );
157     }

159     @Test
160     void copiesExistingNode() throws Exception {
161         final Document dom = DocumentBuilderFactory.newInstance()
162             .newDocumentBuilder().newDocument();
163         final Iterable<Directive> copy = Directives.copyOf(
164             new XMLDocument(
165                 StringUtils.join(
166                     "<jeff name='Jeffrey'><first/><second/>",
167                     "<?some-pi test?>",
168                     "<file a='x'><f><name>\u20ac</name></f></file>",
169                     "<!-- some comment -->",
170                     "<x><![CDATA[hey you]]></x>  </jeff>"
171                 )
172             ).node()
173         );
174         MatcherAssert.assertThat(copy, Matchers.iterableWithSize(19));
175         new Xembler(new Directives().add("dudes").append(copy)).apply(dom);
176         MatcherAssert.assertThat(
177             XhtmlMatchers.xhtml(dom),
178             XhtmlMatchers.hasXPaths(
179                 "/dudes/jeff[@name = 'Jeffrey']",
180                 "/dudes/jeff[first and second]",
181                 "/dudes/jeff/file[@a='x']/f[name='\u20ac']"
182             )
183         );
184     }

272     @Test
273     void acceptsFromMultipleThreads() throws Exception {
274         final Directives dirs = new Directives().add("mt6");
275         final int tasks = 50;
276         new LengthOf(
277             new Threads<>(
278                 tasks / 10,
279                 new Repeated<>(
280                     () -> {
281                         dirs.append(
282                             new Directives()
283                                 .add("fo9").attr("yu", "").set("some text 90").up()
284                                 .add("tr4").attr("s2w3", "").set("some other text 76")
285                                 .up()
286                         );
287                         return null;
288                     },
289                     tasks
290                 )
291             )
292         ).value();
293         MatcherAssert.assertThat(
294             XhtmlMatchers.xhtml(new Xembler(dirs).xml()),
295             XhtmlMatchers.hasXPath("/mt6[count(fo9[@yu])=50]")
296         );
297     }
