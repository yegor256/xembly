30  package org.xembly;
31  
32  import lombok.EqualsAndHashCode;
33  
34  /**
35   * Argument properly escaped.
36   *
37   * <p>The class is immutable and thread-safe.
38   *
39   * @author Yegor Bugayenko (yegor@teamed.io)
40   * @version $Id: b0189f370c3c9e507e6d896df89131fbda841362 $
41   * @since 0.1
42   */
43  @EqualsAndHashCode(of = "value")
44  final class Arg {
45  
46      /**
47       * Value of it.
48       */
49      private final transient String value;
50  
51      /**
52       * Public ctor.
53       * @param val Value of it
54       * @throws XmlContentException If fails
55       */
56      Arg(final String val) throws XmlContentException {
57          for (final char chr : val.toCharArray()) {
58              Arg.legal(chr);
59          }
60          this.value = val;
61      }
62  
63      @Override
64      public String toString() {
65          final String escaped = Arg.escape(this.value);
66          return new StringBuilder(this.value.length() + 2 + escaped.length())
67              .append('"').append(escaped).append('"').toString();
68      }
69  
70      /**
71       * Get it's raw value.
72       * @return Value
73       */
74      public String raw() {
75          return this.value;
76      }
77  
78      /**
79       * Un-escape all XML symbols.
80       * @param text XML text
81       * @return Clean text
82       * @throws XmlContentException If fails
83       */
84      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
85      public static String unescape(final String text)
86          throws XmlContentException {
87          final char[] chars = text.toCharArray();
88          if (chars.length < 2) {
89              throw new IllegalArgumentException(
90                  "internal error, argument can't be shorter than 2 chars"
91              );
92          }
93          final int len = chars.length - 1;
94          final StringBuilder output = new StringBuilder(text.length());
95          for (int idx = 1; idx < len; ++idx) {
96              if (chars[idx] == '&') {
97                  final StringBuilder sbuf = new StringBuilder(0);
98                  while (chars[idx] != ';') {
99                      // @checkstyle ModifiedControlVariable (1 line)
100                     ++idx;
101                     if (idx == chars.length) {
102                         throw new XmlContentException(
103                             "reached EOF while parsing XML symbol"
104                         );
105                     }
106                     sbuf.append(chars[idx]);
107                 }
108                 output.append(Arg.symbol(sbuf.substring(0, sbuf.length() - 1)));
109             } else {
110                 output.append(chars[idx]);
111             }
112         }
113         return output.toString();
114     }
115 
116     /**
117      * Escape all unprintable characters.
118      * @param text Raw text
119      * @return Clean text
120      */
121     private static String escape(final String text) {
122         final StringBuilder output = new StringBuilder(text.length());
123         for (final char chr : text.toCharArray()) {
124             if (chr < ' ') {
125                 output.append("&#").append((int) chr).append(';');
126             } else if (chr == '"') {
127                 output.append("&quot;");
128             } else if (chr == '&') {
129                 output.append("&amp;");
130             } else if (chr == '\'') {
131                 output.append("&apos;");
132             } else if (chr == '<') {
133                 output.append("&lt;");
134             } else if (chr == '>') {
135                 output.append("&gt;");
136             } else {
137                 output.append(chr);
138             }
139         }
140         return output.toString();
141     }
142 
143     /**
144      * Convert XML symbol to char.
145      * @param symbol XML symbol, without leading ampersand
146      * @return Character
147      * @throws XmlContentException If fails
148      */
149     private static char symbol(final String symbol) throws XmlContentException {
150         final char chr;
151         if ('#' == symbol.charAt(0)) {
152             final int num = Integer.parseInt(symbol.substring(1));
153             chr = Arg.legal((char) num);
154         } else if ("apos".equalsIgnoreCase(symbol)) {
155             chr = '\'';
156         } else if ("quot".equalsIgnoreCase(symbol)) {
157             chr = '"';
158         } else if ("lt".equalsIgnoreCase(symbol)) {
159             chr = '<';
160         } else if ("gt".equalsIgnoreCase(symbol)) {
161             chr = '>';
162         } else if ("amp".equalsIgnoreCase(symbol)) {
163             chr = '&';
164         } else {
165             throw new XmlContentException(
166                 String.format("unknown XML symbol &%s;", symbol)
167             );
168         }
169         return chr;
170     }
171 
172     /**
173      * Validate char number and throw exception if it's not legal.
174      * @param chr Char number
175      * @return The same number
176      * @throws XmlContentException If illegal
177      */
178     private static char legal(final char chr) throws XmlContentException {
179         // @checkstyle MagicNumber (5 lines)
180         Arg.range(chr, 0x00, 0x08);
181         Arg.range(chr, 0x0B, 0x0C);
182         Arg.range(chr, 0x0E, 0x1F);
183         Arg.range(chr, 0x7F, 0x84);
184         Arg.range(chr, 0x86, 0x9F);
185         return chr;
186     }
187 
188     /**
189      * Throw if number is in the range.
190      * @param chr Char number
191      * @param left Left number (inclusive)
192      * @param right Right number (inclusive)
193      * @throws XmlContentException If illegal
194      */
195     private static void range(final char chr, final int left, final int right)
196         throws XmlContentException {
197         if (chr >= left && chr <= right) {
198             throw new XmlContentException(
199                 String.format(
200                     // @checkstyle LineLength (1 line)
201                     "Character #%02X is in the restricted XML range #%02X-#%02X, see http://www.w3.org/TR/2004/REC-xml11-20040204/#charsets",
202                     (int) chr, left, right
203                 )
204             );
205         }
206     }
